import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;
public class Medium2 {
	public static void main(String[] args) {
		try {
			Scanner scanner = new Scanner(System.in);

			List<String> inputList = new ArrayList<>();
			String line;

			while (scanner.hasNext()) {
				line = scanner.nextLine();
				inputList.add(line);
			}

			scanner.close();
			
			String[] strSplit = inputList.get(0).split(" ");
	
			int index = Integer.parseInt(strSplit[0]);
			
			
			Integer [] numbers = new Integer[index];
			int location = 0;
			int addedNum = 0;
			
			for (int i=0; i < index;i++) {
				String[] strSplit2 = inputList.get(i+1).split(" ");
				location = Integer.parseInt(strSplit2[0]);
				addedNum = Integer.parseInt(strSplit2[1]);
				numbers[location] = numbers[location] + addedNum;  
			}
			System.out.println(numbers);

		} catch (Exception e) {
			e.printStackTrace();
		}

		}
}
