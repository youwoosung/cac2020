import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class 3medium {
	public static void main(String[] args) {
		try {
			Scanner scanner = new Scanner(System.in);

			List<String> inputList = new ArrayList<>();
			String line;

			while (scanner.hasNext()) {
				line = scanner.nextLine();
				inputList.add(line);
			}
			
			scanner.close();
			
			for (int i = 0; i < inputList.length(); i = i + 3) {
			String code = inputList.get(0).substring(i, i + 3);
				if (code.equals("LLL")) {
				code = "A";
			} else if (code.equals("SSL")){
				code = "T";
			} else if(code.equals("SLL")){
				code = "G";
			} else if (code.equals("SLS")){
				code = "C";
			}
				System.out.print(code);
			}
			


		} catch (Exception e) {
			e.printStackTrace();
	}

}
}
